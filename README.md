# get start on docker

1. apt-get install npm
2. npm install -g gulp
3. ln -s /usr/bin/nodejs /usr/bin/node
4. npm install semantic-ui --save
5. cd semantic
6. gulp build
7. ln -s semantic public/semantic
8. npm start
